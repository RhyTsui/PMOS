🔴 三、Redis 任务队列（核心）
redis_queue.py
import redis
import json

r = redis.Redis(host="localhost", port=6379, decode_responses=True)

QUEUE_KEY = "agent_tasks"

def push_task(task: dict):
    r.lpush(QUEUE_KEY, json.dumps(task))

def pop_task():
    task = r.rpop(QUEUE_KEY)
    if task:
        return json.loads(task)
    return None

def queue_size():
    return r.llen(QUEUE_KEY)
⚡ 四、Worker Pool（10~50 并行 Claude）
worker_pool.py
import asyncio
from agents.worker import WorkerAgent
from redis_queue import pop_task

class WorkerPool:

    def __init__(self, size=20):
        self.size = size
        self.workers = [WorkerAgent(worker_id=i) for i in range(size)]

    async def run_worker(self, worker):

        while True:
            task = pop_task()

            if not task:
                await asyncio.sleep(2)
                continue

            print(f"[Worker {worker.worker_id}] executing {task['id']}")

            result = worker.run(task)

            worker.save_result(task, result)

    async def start(self):
        await asyncio.gather(
            *[self.run_worker(w) for w in self.workers]
        )

🧠 六、LangGraph 编排（v2核心升级点）
workflow.py
from langgraph.graph import StateGraph
from agents.planner import PlannerAgent
from agents.reviewer import ReviewerAgent
from agents.synthesizer import SynthesizerAgent

def build_graph():

    graph = StateGraph(dict)

    planner = PlannerAgent()
    reviewer = ReviewerAgent()
    synthesizer = SynthesizerAgent()

    def plan_node(state):
        return planner.create_tasks(state)

    def review_node(state):
        return reviewer.review(state)

    def synth_node(state):
        return synthesizer.merge(state)

    graph.add_node("plan", plan_node)
    graph.add_node("review", review_node)
    graph.add_node("synth", synth_node)

    graph.set_entry_point("plan")
    graph.add_edge("plan", "review")
    graph.add_edge("review", "synth")

    return graph.compile()
