🧩 四、关键词自进化（工程版）
crawler/keyword_evolution.py
from collections import Counter
import redis
import json

r = redis.Redis(host='redis', port=6379)

STOPWORDS = {"的", "了", "啊", "吗"}

def extract_keywords(texts):

    words = []

    for t in texts:
        words.extend(t.split())

    counter = Counter(words)

    new_keywords = [
        w for w, c in counter.items()
        if c > 5 and w not in STOPWORDS
    ]

    return new_keywords


def update_keyword_pool(new_keywords):

    for kw in new_keywords:
        r.sadd("keywords", kw)
🧠 五、异步队列（核心）
queue/redis_queue.py
import redis
import json

r = redis.Redis(host='redis', port=6379)


def push_video(task):
    r.lpush("video_queue", json.dumps(task))


def pop_video():
    task = r.rpop("video_queue")
    if task:
        return json.loads(task)
    return None
🕷️ 六、采集 → 队列（生产级写法）
crawler/douyin_pw.py（核心改造）
from playwright.sync_api import sync_playwright
from queue.redis_queue import push_video


def crawl(keyword):

    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True)
        page = browser.new_page()

        page.goto(f"https://www.douyin.com/search/{keyword}")

        page.wait_for_timeout(3000)

        videos = page.query_selector_all("video")

        for v in videos:
            src = v.get_attribute("src")

            if src:
                push_video({
                    "video_url": src,
                    "keyword": keyword
                })

        browser.close()
🎬 七、解析Worker（核心）
app/worker.py
from queue.redis_queue import pop_video
from pipeline.download import download
from pipeline.asr import transcribe
from pipeline.ocr import extract_text
from processor.structurer import structure
from storage.db import save


def run_worker():

    while True:

        task = pop_video()

        if not task:
            continue

        video_path = download(task["video_url"])

        asr_text = transcribe(video_path)
        ocr_text = extract_text(video_path)

        data = structure(asr_text, ocr_text)

        save(data)
🧠 八、主入口（调度）
app/main.py
from crawler.keyword_engine import generate_keywords
from crawler.keyword_evolution import extract_keywords, update_keyword_pool
from crawler.douyin_pw import crawl
import threading


def run():

    keywords = generate_keywords()

    # 启动worker线程
    for _ in range(3):
        t = threading.Thread(target=run_worker)
        t.start()

    # 主循环采集
    for kw in keywords:
        crawl(kw)


if __name__ == "__main__":
    run()
