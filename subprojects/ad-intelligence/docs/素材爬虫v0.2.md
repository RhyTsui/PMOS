def crawl_feed(page):

    for _ in range(20):

        # 模拟滑动
        page.mouse.wheel(0, 3000)

        # 停留（模拟用户）
        page.wait_for_timeout(2000)

        videos = page.query_selector_all("video")

        for v in videos:
            src = v.get_attribute("src")
            if src:
                push_video(src)
