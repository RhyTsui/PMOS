    5. validator.py 参数校验模块
    from typing import Dict, List, Tuple
from config import ParamConfig
from logger import get_log_adapter

def validate_params(
    params: Dict,
    param_configs: List[ParamConfig],
    call_type: str
) -> Tuple[bool, str, Dict]:
    """
    统一参数校验
    返回：(校验是否通过, 错误信息, 清洗后的参数字典)
    """
    log = get_log_adapter(call_type)
    cleaned_params = {}
    required_params = [p for p in param_configs if p.required]

    # 校验必填参数
    missing_params = []
    for p in required_params:
        if p.name not in params or params[p.name] is None or str(params[p.name]).strip() == "":
            missing_params.append(p.name)
    if missing_params:
        error_msg = f"必填参数缺失：{', '.join(missing_params)}"
        log.error(error_msg)
        return False, error_msg, {}

    # 校验参数类型 & 清洗参数
    for p in param_configs:
        if p.name not in params or params[p.name] is None:
            continue
        param_value = params[p.name]
        # 简单类型校验
        if p.type == "string":
            cleaned_params[p.name] = str(param_value).strip()
        elif p.type == "number":
            try:
                cleaned_params[p.name] = float(param_value)
            except ValueError:
                error_msg = f"参数{p.name}类型错误，要求数字类型，实际值：{param_value}"
                log.error(error_msg)
                return False, error_msg, {}
        elif p.type == "integer":
            try:
                cleaned_params[p.name] = int(param_value)
            except ValueError:
                error_msg = f"参数{p.name}类型错误，要求整数类型，实际值：{param_value}"
                log.error(error_msg)
                return False, error_msg, {}
        else:
            cleaned_params[p.name] = param_value

    log.info(f"参数校验通过，清洗后参数：{cleaned_params}")
    return True, "", cleaned_params

    import httpx
from typing import Dict, Tuple
from config import get_config
from logger import get_log_adapter

config = get_config()
bi_config = config.bi

async def call_bi_api(
    api_path: str,
    method: str,
    params: Dict,
    call_type: str
) -> Tuple[bool, str, Dict]:
    """
    统一调用BI接口
    返回：(是否成功, 错误信息, 返回数据)
    """
    log = get_log_adapter(call_type)
    url = f"{bi_config.base_url.rstrip('/')}/{api_path.lstrip('/')}"
    headers = {
        "Authorization": f"Bearer {bi_config.auth_token}",
        "Content-Type": "application/json"
    }

    log.info(f"开始调用BI接口：{method} {url}，入参：{params}")
    try:
        async with httpx.AsyncClient(timeout=bi_config.timeout) as client:
            if method.upper() == "GET":
                response = await client.get(url, params=params, headers=headers)
            elif method.upper() == "POST":
                response = await client.post(url, json=params, headers=headers)
            else:
                error_msg = f"不支持的请求方法：{method}"
                log.error(error_msg)
                return False, error_msg, {}

            response.raise_for_status()
            result = response.json()
            log.info(f"BI接口调用成功，返回数据：{result}")
            return True, "", result

    except httpx.TimeoutException:
        error_msg = f"BI接口调用超时，超时时间：{bi_config.timeout}秒"
        log.error(error_msg)
        return False, error_msg, {}
    except httpx.HTTPStatusError as e:
        error_msg = f"BI接口调用失败，状态码：{e.response.status_code}，响应内容：{e.response.text}"
        log.error(error_msg)
        return False, error_msg, {}
    except Exception as e:
        error_msg = f"BI接口调用异常：{str(e)}"
        log.error(error_msg)
        return False, error_msg, {}

        7. cli_parser.py CLI 命令解析模块
        import shlex
from typing import Dict, Tuple
from config import get_config, BusinessToolConfig
from logger import get_log_adapter

config = get_config()
# 命令名 -> 工具配置的映射
command_map = {tool.command_name: tool for tool in config.business_tools}

def parse_cli_command(command_str: str) -> Tuple[bool, str, BusinessToolConfig, Dict]:
    """
    解析CLI命令字符串
    返回：(是否成功, 错误信息, 对应的工具配置, 解析后的参数字典)
    """
    log = get_log_adapter("CLI-like")
    log.info(f"开始解析CLI命令：{command_str}")

    try:
        # 拆分命令，处理带引号的参数值
        parts = shlex.split(command_str.strip())
        if not parts:
            error_msg = "命令不能为空"
            log.error(error_msg)
            return False, error_msg, None, {}

        # 提取命令名
        command_name = parts[0]
        if command_name not in command_map:
            error_msg = f"未知命令：{command_name}，支持的命令：{', '.join(command_map.keys())}"
            log.error(error_msg)
            return False, error_msg, None, {}

        tool_config = command_map[command_name]
        params = {}

        # 解析--key value格式参数
        i = 1
        while i < len(parts):
            part = parts[i]
            if part.startswith("--"):
                key = part.lstrip("-")
                if i + 1 >= len(parts) or parts[i+1].startswith("--"):
                    error_msg = f"参数--{key}缺少对应的值"
                    log.error(error_msg)
                    return False, error_msg, None, {}
                value = parts[i+1]
                params[key] = value
                i += 2
            else:
                error_msg = f"不支持的参数格式：{part}，请使用--key value格式"
                log.error(error_msg)
                return False, error_msg, None, {}

        log.info(f"CLI命令解析成功，命令名：{command_name}，解析后参数：{params}")
        return True, "", tool_config, params

    except Exception as e:
        error_msg = f"CLI命令解析异常：{str(e)}"
        log.error(error_msg)
        return False, error_msg, None, {}

# ==================== MCP协议标准模型 ====================
class MCPToolParameter(BaseModel):
    type: str
    description: str
    required: Optional[bool] = None

class MCPTool(BaseModel):
    name: str
    description: str
    inputSchema: Dict[str, Any]

class MCPListToolsResponse(BaseModel):
    tools: List[MCPTool]

class MCPToolCallRequest(BaseModel):
    name: str
    arguments: Dict[str, Any]

class MCPToolCallResponse(BaseModel):
    content: List[Dict[str, Any]]
    isError: bool = False

# ==================== 工具配置转换 ====================
def tool_to_mcp_format(tool: BusinessToolConfig) -> MCPTool:
    """将业务工具配置转换为MCP标准工具格式"""
    properties = {}
    required = []
    for param in tool.params:
        properties[param.name] = {
            "type": param.type,
            "description": param.description
        }
        if param.required:
            required.append(param.name)
    return MCPTool(
        name=tool.tool_name,
        description=tool.description,
        inputSchema={
            "type": "object",
            "properties": properties,
            "required": required
        }
    )

def get_cli_mcp_tool() -> MCPTool:
    """生成CLI-like的单工具MCP格式"""
    # 生成命令说明
    command_desc = []
    for tool in config.business_tools:
        param_desc = " ".join([f"--{p.name} {p.name.upper()}" for p in tool.params])
        command_desc.append(f"- {tool.command_name} {param_desc}：{tool.description}")
    command_desc_str = "\n".join(command_desc)

    description = f"""
执行游戏数据查询CLI命令，仅支持以下命令，严格按照格式输出：
{command_desc_str}

格式要求：
1. 命令名 + 空格 + --参数名 参数值
2. 必填参数必须填写，可选参数可省略
3. 带空格的参数值请用引号包裹
4. 一次仅输出一条命令，不要多余内容

示例：
query_dau --game_id 1001 --start_date 2026-04-01 --end_date 2026-04-07
"""
    return MCPTool(
        name="cli",
        description=description.strip(),
        inputSchema={
            "type": "object",
            "properties": {
                "command": {
                    "type": "string",
                    "description": "CLI查询命令字符串"
                }
            },
            "required": ["command"]
        }
    )

# ==================== 原生MCP端点 ====================
@app.post("/mcp/list-tools", response_model=MCPListToolsResponse, tags=["原生MCP"])
async def mcp_list_tools():
    """原生MCP：获取工具清单"""
    log = get_log_adapter("原生MCP")
    log.info("收到工具清单请求")
    tools = [tool_to_mcp_format(tool) for tool in config.business_tools]
    return MCPListToolsResponse(tools=tools)

@app.post("/mcp/call-tool", response_model=MCPToolCallResponse, tags=["原生MCP"])
async def mcp_call_tool(request: MCPToolCallRequest):
    """原生MCP：执行工具调用"""
    call_type = "原生MCP"
    log = get_log_adapter(call_type)
    tool_name = request.name
    arguments = request.arguments
    log.info(f"收到工具调用请求，工具名：{tool_name}，入参：{arguments}")

    # 查找工具配置
    tool_config = next((t for t in config.business_tools if t.tool_name == tool_name), None)
    if not tool_config:
        error_msg = f"未知工具：{tool_name}"
        log.error(error_msg)
        return MCPToolCallResponse(
            content=[{"type": "text", "text": error_msg}],
            isError=True
        )

    # 参数校验
    is_valid, error_msg, cleaned_params = validate_params(arguments, tool_config.params, call_type)
    if not is_valid:
        return MCPToolCallResponse(
            content=[{"type": "text", "text": error_msg}],
            isError=True
        )

    # 调用BI接口
    is_success, error_msg, result = await call_bi_api(
        api_path=tool_config.bi_api_path,
        method=tool_config.request_method,
        params=cleaned_params,
        call_type=call_type
    )

    if not is_success:
        return MCPToolCallResponse(
            content=[{"type": "text", "text": error_msg}],
            isError=True
        )

    # 返回标准化结果
    return MCPToolCallResponse(
        content=[{"type": "text", "text": f"查询成功，结果：{result}"}]
    )

## 保存方法
1. 新建4个空白的文本文档
2. 分别把上面4个代码块的内容完整复制进去
3. 分别保存为对应的文件名（注意后缀是`.md`，编码选`UTF-8`）
4. 保存后你就有了4个独立的Markdown文档，可直接在Claude Code中打开使用。
