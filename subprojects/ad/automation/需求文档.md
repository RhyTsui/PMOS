3️⃣ Sankey 图组件（核心）
// src/components/SankeyChart.jsx

import React, { useEffect, useRef } from "react";
import * as echarts from "echarts";

export default function SankeyChart({ data, onNodeClick }) {
  const ref = useRef();

  useEffect(() => {
    const chart = echarts.init(ref.current);

    const option = {
      tooltip: {},
      series: [
        {
          type: "sankey",
          data: data.nodes,
          links: data.links,
          emphasis: {
            focus: "adjacency"
          }
        }
      ]
    };

    chart.setOption(option);

    chart.on("click", function (params) {
      if (params.dataType === "node") {
        onNodeClick(params.name);
      }
    });

    return () => chart.dispose();
  }, [data]);

  return <div ref={ref} style={{ width: "100%", height: "500px" }} />;
}
4️⃣ 页面主逻辑
// src/pages/FlowPage.jsx

import React, { useEffect, useState } from "react";
import { getFlow, getUsers, getTrace } from "../api";
import SankeyChart from "../components/SankeyChart";

export default function FlowPage() {
  const [flow, setFlow] = useState(null);
  const [users, setUsers] = useState([]);
  const [trace, setTrace] = useState([]);

  useEffect(() => {
    getFlow().then(res => setFlow(res.data));
  }, []);

  const handleNodeClick = (stage) => {
    getUsers(stage).then(res => setUsers(res.data.users));
  };

  const handleUserClick = (userId) => {
    getTrace(userId).then(res => setTrace(res.data.trace));
  };

  if (!flow) return <div>Loading...</div>;

  return (
    <div>
      <h2>Ads Flow Insight</h2>

      <SankeyChart data={flow} onNodeClick={handleNodeClick} />

      <h3>Users</h3>
      {users.map(u => (
        <div key={u.user_id} onClick={() => handleUserClick(u.user_id)}>
          {u.user_id} - {u.device_id}
        </div>
      ))}

      <h3>Trace</h3>
      {trace.map((t, i) => (
        <div key={i}>{t.step} - {t.time}</div>
      ))}
    </div>
  );
}
